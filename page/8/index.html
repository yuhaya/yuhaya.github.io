namespace EtahMvc;

        }
        return $this->inputFilter;
    }
    /*
      可传入需要过滤的字段(必须为字符串类型参数)
      @param  string $name
      @param  string $name
      .......
      @return boolean
     */
    public function isValid(){
        $this->setInputFilter($this->getInputFilter());
        if(func_num_args() != 0){
            $this->setValidationGroup(func_get_args());
        }

/*
  Sets one or multiple options
 
  @param  array|Traversable $options Options to set
  @throws ExceptionInvalidArgumentException If $options is not an array or Traversable
  @return AbstractValidator Provides fluid interface
 */
publicfunctionsetOptions($options=array())
{
    if(!isarray($options) && !$optionsinstanceofTraversable) {
        thrownewExceptionInvalidArgumentException(_METHOD.' expects an array or Traversable');
    }

    foreach($optionsas$name=>$option) {
        $fname='set'. ucfirst($name);
        $fname2='is'. ucfirst($name);
        if(($name!='setOptions') && method_exists($this,$name)) {
            $this->{$name}($option);
        }elseif(($fname!='setOptions') && method_exists($this,$fname)) {
            $this->{$fname}($option);
        }elseif(method_exists($this,$fname2)) {
            $this->{$fname2}($option);
        }elseif(isset($this->options)) {
            $this->options[$name] =$option;
        }else{
            $this->abstractOptions[$name] =$option;
        }
    }

    return$this;
}

Zend Framework 2 : Working with AuthenticationService and Session Db Save Handler

One of the Session SaveHandlers that Zend Framework 2 provide is DbTableGateway Save Handler that utilize ZendDb as a session save handler. How to combine authentication service with it ? Let’s learn about it! Save Handlers themselves are decoupled from PHP’s save handler functions and are only implemented as a PHP save handler when utilized in conjunction with ZendSessionSessionManager.

//filename : module/SanAuthWithDbSaveHandler/src/SanAuthWithDbSaveHandler/Storage/AuthStorage.php
namespaceSanAuthWithDbSaveHandlerStorage;


useZendServiceManagerServiceLocatorAwareInterface;
useZendServiceManagerServiceLocatorInterface;
useZendAuthenticationStorage;
useZendSessionConfigSessionConfig;
useZendDbTableGatewayTableGateway;
useZendSessionSaveHandlerDbTableGateway;
useZendSessionSaveHandlerDbTableGatewayOptions;


classAuthStorageextendsStorageSession
    implementsServiceLocatorAwareInterface
{
    protected$serviceLocator;
    protected$namespace;


publicfunction__construct($namespace= null)
{
    parent::__construct($namespace);

    $this-&gt;namespace=$namespace;
}

publicfunctionsetDbHandler()
{
    $tableGateway=newTableGateway('session',
                                        $this-&gt;getServiceLocator()
                                             -&gt;get('ZendDbAdapterAdapter'));

    $saveHandler=newDbTableGateway($tableGateway,
                                        newDbTableGatewayOptions());

    //open session
    $sessionConfig=newSessionConfig();
    $saveHandler
        -&gt;open($sessionConfig-&gt;getOption('save_path'),$this-&gt;namespace);
   //set save handler with configured session
   $this-&gt;session-&gt;getManager()-&gt;setSaveHandler($saveHandler);
}

publicfunctionwrite($contents)
{
    parent::write($contents);
    /**
        when $this-&gt;authService-&gt;authenticate(); is valid, the session
        automatically called write('username')
      in this case, i want to save data like
     ["storage"] =&gt; array(4) {
          ["id"] =&gt; string(1) "1"
          ["username"] =&gt; string(5) "admin"
          ["ip_address"] =&gt; string(9) "127.0.0.1"
          ["user_agent"] =&gt; string(81) "Mozilla/5.0 (Macintosh; Intel Mac OS X 10.7;
            rv:18.0)
          Gecko/20100101    Firefox/18.0"
    }*/
    if(is_array($contents) &amp;&amp; !empty($contents)) {
        $this-&gt;getSessionManager()
                      -&gt;getSaveHandler()
                      -&gt;write($this-&gt;getSessionId(), ZendJsonJson::encode($contents));
    }
}

publicfunctionclear()
{
    $this-&gt;getSessionManager()-&gt;getSaveHandler()-&gt;destroy($this-&gt;getSessionId());
    parent::clear();
}

publicfunctiongetSessionManager()
{
    return$this-&gt;session-&gt;getManager();
}

publicfunctiongetSessionId()
{
    return$this-&gt;session-&gt;getManager()-&gt;getId();
}

publicfunctionsetServiceLocator(ServiceLocatorInterface$serviceLocator)
{
    $this-&gt;serviceLocator =$serviceLocator;
}

publicfunctiongetServiceLocator()
{
    return$this-&gt;serviceLocator;
}

}

//filename : module/SanAuthWithDbSaveHandler/src/SanAuthWithDbSaveHandler/Controller/AuthController.php
namespaceSanAuthWithDbSaveHandlerController;


useZendMvcControllerAbstractActionController;
useZendAuthenticationAuthenticationService;
useZendViewModelViewModel;


classAuthControllerextendsAbstractActionController
{
    protected$authService;


//we will inject authService via factory
publicfunction__construct(AuthenticationService$authService)
{
    $this-&gt;authService =$authService;
}

publicfunctionindexAction()
{
    if($this-&gt;authService-&gt;getStorage()-&gt;getSessionManager()
             -&gt;getSaveHandler()
             -&gt;read($this-&gt;authService-&gt;getStorage()-&gt;getSessionId())) {
        //redirect to success controller...
        return$this-&gt;redirect()-&gt;toRoute('success');
    }

    $form=$this-&gt;getServiceLocator()
                 -&gt;get('FormElementManager')
                 -&gt;get('SanAuthWithDbSaveHandlerFormLoginForm');
    $viewModel=newViewModel();

    //initialize error...
    $viewModel-&gt;setVariable('error','');
    //authentication block...
    $this-&gt;authenticate($form,$viewModel);

    $viewModel-&gt;setVariable('form',$form);
    return$viewModel;
}

/** this function called by indexAction to reduce complexity of function */
protectedfunctionauthenticate($form,$viewModel)
{
    $request=$this-&gt;getRequest();
    if($request-&gt;isPost()) {
        $form-&gt;setData($request-&gt;getPost());
        if($form-&gt;isValid()) {
            $dataform=$form-&gt;getData();

            $this-&gt;authService-&gt;getAdapter()
                                   -&gt;setIdentity($dataform['username'])
                                   -&gt;setCredential($dataform['password']);
            $result=$this-&gt;authService-&gt;authenticate();
            if($result-&gt;isValid()) {
                //authentication success
                $resultRow=$this-&gt;authService-&gt;getAdapter()-&gt;getResultRowObject();

                $this-&gt;authService-&gt;getStorage()-&gt;write(
                     array('id'         =&gt;$resultRow-&gt;id,
                            'username'  =&gt;$dataform['username'],
                            'ip_address'=&gt;$this-&gt;getRequest()-&gt;getServer('REMOTE_ADDR'),
                            'user_agent'   =&gt;$request-&gt;getServer('HTTP_USER_AGENT'))
                );

                return$this-&gt;redirect()-&gt;toRoute('success',array('action'=&gt;'index'));;
            }else{
                $viewModel-&gt;setVariable('error','Login Error');
            }
        }
    }
}

publicfunctionlogoutAction()
{
    $this-&gt;authService-&gt;getStorage()-&gt;clear();
    return$this-&gt;redirect()-&gt;toRoute('auth');
}

Zend Framework 2 : Using HydratingResultSet and ObjectProperty’s Hydrator

When joining table using ZendDb, calling coloumn(s) in other table when make a join is a pain. It because if we use normal ZendDbResultSetResultSet and setting up ArrayObjectPrototype with the class object, we only get the current model class only ( that consist of columns of single table). Register other table column on that class is NOT a solution, it will make a confusion on what entity is for. Maybe not best (errr…good) solution but simplest solution is to use ZendDbResultSetHydratingResultSet and ZendStdlibHydratorObjectProperty. 

There is I explain on case situation : “we have a table named album ( with columns : `id`, `artist`, `title` ) and table named track ( with columns : `track_id`, `track_title`, and `album_id` as foreign key agains album )”.

Ok, let’s create model like as the docs :

    {
        $this->id     = (isset($data['id']))     ? $data['id']     : null;
        $this->artist = (isset($data['artist'])) ? $data['artist'] : null;
        $this->title  = (isset($data['title']))  ? $data['title']  : null;
    }

    public function getArrayCopy()
    {
        return get_object_vars($this);
    }

zf2多模块配置不同的layout

zf2的view_manager的配置，设计的真是一朵奇葩，我实在搞不清他们为什么这么设计。想必大部分的人都是从ZendSkeletonApplication开始学习zf2的，也理所当然的认为每个module的view_manager是独立的，互补干扰的，但是我们错了，不同module之间共用同一个view_manager配置，后来加载module的view_manager会覆盖前面加载的module的view_manager同名（即key值相同）配置，这种设计大部分时候都没什么问题，但是像layout这种配置就开始互相覆盖了，好了，不废话了，开始介绍下怎么为不同模块配置不同的layout。

1.新建一个Test模块（将Application模块复制一份，改改相关的名字即可），在zf2.local/module/Test/config/module.config.php的view_manager的template_map配置你module的layout路径，代码如下

'view_manager'=>array(
'template_map'=>array(
'test/layout'=> __DIR__ .'/../view/test/layout/layout.phtml',
),
'template_path_stack'=>array(
'test'=>__DIR__ .'/../view/'
),
)
