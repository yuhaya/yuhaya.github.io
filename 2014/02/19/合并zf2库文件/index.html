<?php
/**
 * This makes our life easier when dealing with paths. Everything is relative
 * to the application root now.
 */
header('Content-Type: text/html; charset=utf-8');
date_default_timezone_set('PRC');
chdir(dirname(__DIR__));
define('__ROOT__', dirname(__DIR__));
define('BASE_PATH', __DIR__);
define('TEMP_PATH',dirname(__DIR__).'/media/temp/');
define('DEVELOP_ENV',0);
if(DEVELOP_ENV){
    error_reporting(E_ALL);
    ini_set('display_errors', 1);
}else{
    ini_set('display_errors', 0);
}

function all_files($file_pos = 'all_file.php'){
    if(file_exists($file_pos)){
        return;
    }
    $file = get_included_files();
    $black_tip = array(
        'SplAutoloader.php'
    );
    foreach ($file as $k=>$value){
        foreach ($black_tip as $v){
            if(preg_match('/'.$v.'/', $value)){
                unset($file[$k]);
            }
        }
    }
    $str = "<?php\n return ".var_export($file,true).";\n?>";
    file_put_contents($file_pos, $str);
}
function strip_whitespace($content) {
    $stripStr = '';
    $tokens =   token_get_all ($content);
    $last_space = false;
    for ($i = 0, $j = count ($tokens); $i < $j; $i++){
        if (is_string ($tokens[$i])){
            $last_space = false;
            $stripStr .= $tokens[$i];
        }
        else{
            switch ($tokens[$i][0]){
                case T_COMMENT:
                case T_DOC_COMMENT:
                case T_OPEN_TAG;
                case T_CLOSE_TAG;
                    break;
                case T_WHITESPACE:
                    if (!$last_space){
                        $stripStr .= ' ';
                        $last_space = true;
                    }
                    break;
                default:
                    $last_space = false;
                    $stripStr .= $tokens[$i][1];
            }
        }
    }
    return $stripStr;
}
function complie($arr_file,$file_name,$extra_file){
    if(!file_exists($file_name)){
        file_put_contents($file_name, '');
    }
    if(file_get_contents($file_name) != ''){
        file_put_contents($file_name, '');
    }
    if(file_get_contents($extra_file) != ''){
        file_put_contents($extra_file, '');
    }
    $run_file = array(
            $file_name,
            $extra_file
        );
    foreach ($arr_file as $key => $value) {
        if(in_array($value, $run_file)){
            continue;
        }
        $res = strip_whitespace(file_get_contents($value));
        $res = preg_replace('/^\s*/','', $res);//匹配开始空格
        $dir = str_replace(DIRECTORY_SEPARATOR.basename($value), '', $value);
        $res = preg_replace('/__DIR__/',"'".$dir."'", $res);//匹配开始空格
        $position = stripos($res,'namespace');
        if($position === 0){
            file_put_contents($file_name, "\n".$res,FILE_APPEND);
        }else{
            file_put_contents($extra_file, $value."\n",FILE_APPEND);
        }
    }

    $complie = file_get_contents($file_name);
    $str = "<?php \n".$complie."\n?>";
    file_put_contents($file_name, $str);
}
function load_complie($complie = "complie.php",$all_file = 'all_file.php',$extra_file = 'extra.txt'){
    if(file_exists($complie)){
        require $complie;
    }elseif(file_exists($all_file)){
        $file_arr = require $all_file;
        complie($file_arr,$complie,$extra_file);
    }
}

if(!DEVELOP_ENV){
    load_complie(__ROOT__.'/cache/complie.php',__ROOT__.'/cache/all_file.php',__ROOT__.'/cache/extra.txt');
}

// Setup autoloading
require 'init_autoloader.php';

// Run the application!
Zend\Mvc\Application::init(require 'config/application.config.php')->run();

all_files(__ROOT__.'/cache/all_file.php');
